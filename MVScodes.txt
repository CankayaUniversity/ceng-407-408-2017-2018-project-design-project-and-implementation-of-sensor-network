        private void button2_Click(object sender, EventArgs e)

            //BAÐLANTIYI KES BUTONU
            timer1.Stop();

        private void button1_Click_1(object sender, EventArgs e)
        {
            timer1.Start(); //250 ms olarak ayarlandý timer'ý.
            if (serialPort1.IsOpen == false)
            {
                if (comboBox1.Text == "")
                    return;
                serialPort1.PortName = comboBox1.Text;  // combobox1'e zaten port isimlerini aktarmýþtýk.
                serialPort1.BaudRate = 9600; //Seri Haberleþme baudrate'i combobox2 'de seçilene göre belirliyoruz.
                try
                {
                    serialPort1.Open(); //Haberleþme için port açýlýyor
                    label2.ForeColor = Color.Green;
                    label2.Text = "Baðlantý Açýk";


                }
                catch (Exception hata)
                {
                    MessageBox.Show("Hata:" + hata.Message);
                }
            }
            else
            {
                label2.Text = "Baðlantý kurulu !!!";
            }
        }

        private void timer1_Tick_1(object sender, EventArgs e)
        {
            try
            {
                serialPort1.DiscardInBuffer();
                string sonuc = serialPort1.ReadLine();//Serial.print kodu ile gelen analog veriyi alýyoruz,string formatýnda sonuc'a atýyoruz
                string[] ldr = sonuc.Split('*');
                label1.Text = sonuc + "";

                veri = Int32.Parse(ldr[0]);
                temp = 660;
                if (veri <= temp)
                {
                    value = false;
                    textBox1.BackColor = Color.Red;
                    textBox1.Text = "DOLU";
                }
                else if (veri >= temp)
                {
                    value = true;
                    textBox1.BackColor = Color.Green;
                    textBox1.Text = "BOÞ";
                    
                }
                //ÝKÝNCÝ DEVRE KONTROL KISMI
                veri = Int32.Parse(ldr[1]);
                temp2 = 660;
                if (veri <= temp)
                {
                    value2 = false;
                    textBox2.BackColor = Color.Red;
                    textBox2.Text = "DOLU";
                }
                else if (veri >= temp)
                {
                    value2 = true;
                    textBox2.BackColor = Color.Green;

                    textBox2.Text = "BOÞ";
                    
                }
                label3.Text = value.ToString();
                // Array.Clear(subStrings, 0, subStrings.Length);
            }
            catch (Exception ex)
            {
                MessageBox.Show(ex.Message); // basarýsýz olursa hata verecek.
                timer1.Stop();
            }

        }
